```js Node
import {
  Supadata,
  JobId
} from '@supadata/js';

// Initialize the client
const supadata = new Supadata({
  apiKey: 'YOUR_API_KEY',
});

// Crawl website
const crawl: JobId = await supadata.web.crawl({
  url: 'https://supadata.ai',
  limit: 10,
});

console.log(`Started crawl job: ${crawl.jobId}`);
```
